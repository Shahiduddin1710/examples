SECTION 1 – SQL
(emp, company, proj)

Schema:
emp(eid, ename, address, sal)
company(cid, cname, eid)
proj(pid, pname, hrs, eid)

1️⃣ Display name of employee who stays in Bandra

SELECT ename
FROM emp
WHERE address = 'Bandra';

2️⃣ Modify database such that eid 101 will work in 'Contentstack'

UPDATE company
SET cname = 'Contentstack'
WHERE eid = 101;

3️⃣ Show project name having working hours greater than 10

SELECT pname
FROM proj
WHERE hrs > 10;

4️⃣ Display employee name and project name for each employee

SELECT e.ename, p.pname
FROM emp e
JOIN proj p ON e.eid = p.eid;

5️⃣ Find average hours from project relation

SELECT AVG(hrs) AS avg_hours
FROM proj;

6️⃣ Display company names starting with letter 'A'

SELECT cname
FROM company
WHERE cname LIKE 'A%';

7️⃣ Display employees working in Reliance company

SELECT e.ename
FROM emp e
JOIN company c ON e.eid = c.eid
WHERE c.cname = 'Reliance';

8️⃣ Display employees who do NOT work for ABC company

SELECT e.ename
FROM emp e
WHERE e.eid NOT IN (
    SELECT eid
    FROM company
    WHERE cname = 'ABC'
);

9️⃣ Find employee details who work for more than 2 hours

SELECT e.*
FROM emp e
JOIN proj p ON e.eid = p.eid
WHERE p.hrs > 2;

🔟 Find all employees whose salary is less than Nisha's salary

SELECT *
FROM emp
WHERE sal < (
    SELECT sal
    FROM emp
    WHERE ename = 'Nisha'
);

4. List employees working in department 101

WHERE department_id = 101;

5. Display employees whose salary > 50,000

FROM Employee

6. Find employees whose name starts with 'A'

7. Retrieve employees ordered by salary DESC

SELECT *
FROM Employee
ORDER BY salary DESC;

8. Find employee details where age is between 25 and 35

FROM Employee
