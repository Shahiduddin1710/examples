SQL & Relational Algebra Complete Practice

SECTION 1 – SQL (emp, company, proj)

emp(eid, ename, address, sal)
company(cid, cname, eid)
proj(pid, pname, hrs, eid)
1) Display name of employee who stays in Bandra
SELECT ename FROM emp WHERE address = 'Bandra';
2) Modify database such that eid 101 will work in 'Contentstack'
UPDATE company SET cname = 'Contentstack' WHERE eid = 101;
3) Show project name having working hours greater than 10
SELECT pname FROM proj WHERE hrs > 10;
4) Display employee name and project name for each employee
SELECT e.ename, p.pname FROM emp e JOIN proj p ON e.eid = p.eid;
5) Find average hours from project relation
SELECT AVG(hrs) AS avg_hours FROM proj;
6) Display company names starting with letter 'A'
SELECT cname FROM company WHERE cname LIKE 'A%';
7) Display employees working in Reliance company
SELECT e.ename FROM emp e JOIN company c ON e.eid = c.eid WHERE c.cname = 'Reliance';
8) Display employees who do NOT work for ABC company
SELECT ename FROM emp WHERE eid NOT IN ( SELECT eid FROM company WHERE cname = 'ABC' );
9) Find employee details who work for more than 2 hours
SELECT e.* FROM emp e JOIN proj p ON e.eid = p.eid WHERE p.hrs > 2;
10) Find eid who works for project id 101 AND 110
SELECT eid FROM proj WHERE pid IN (101,110) GROUP BY eid HAVING COUNT(DISTINCT pid) = 2;

SECTION 3 – Employee–Department Practice

Employee(emp_id, emp_name, age, salary, department_id)
Department(department_id, department_name, location)

PART A – Simple SQL Queries

1) Retrieve all details of employees
SELECT * FROM Employee;
2) Display only employee names and salaries
SELECT emp_name, salary FROM Employee;
3) Find employees whose age is greater than 30
SELECT * FROM Employee WHERE age > 30;
4) List employees working in department 101
SELECT * FROM Employee WHERE department_id = 101;
5) Display employees whose salary is greater than 50,000
SELECT * FROM Employee WHERE salary > 50000;
6) Find employees whose name starts with 'A'
SELECT * FROM Employee WHERE emp_name LIKE 'A%';
7) Retrieve employees ordered by salary in descending order
SELECT * FROM Employee ORDER BY salary DESC;
8) Find employee details where age is between 25 and 35
SELECT * FROM Employee WHERE age BETWEEN 25 AND 35;
9) Display distinct department IDs from Employee table
SELECT DISTINCT department_id FROM Employee;
10) Find employee with minimum age
SELECT * FROM Employee WHERE age = (SELECT MIN(age) FROM Employee);

PART B – Aggregate Function Queries

11) Find the total number of employees
SELECT COUNT(*) AS total_employees FROM Employee;
12) Find the average salary of employees
SELECT AVG(salary) AS average_salary FROM Employee;
13) Find the maximum salary
SELECT MAX(salary) AS max_salary FROM Employee;
14) Find the minimum age of employees
SELECT MIN(age) AS minimum_age FROM Employee;
15) Find total salary paid to all employees
SELECT SUM(salary) AS total_salary FROM Employee;
16) Find number of employees in each department
SELECT department_id, COUNT(*) AS employee_count FROM Employee GROUP BY department_id;
17) Find average salary department-wise
SELECT department_id, AVG(salary) AS avg_salary FROM Employee GROUP BY department_id;
18) Find departments having more than 5 employees
SELECT department_id FROM Employee GROUP BY department_id HAVING COUNT(*) > 5;
19) Find department with highest average salary
SELECT department_id FROM Employee GROUP BY department_id HAVING AVG(salary) = ( SELECT MAX(avg_sal) FROM ( SELECT AVG(salary) AS avg_sal FROM Employee GROUP BY department_id ) t );
20) Find total salary of employees whose age is greater than 30
SELECT SUM(salary) FROM Employee WHERE age > 30;

PART C – GROUP BY + HAVING Practice

21) Display department-wise employee count
SELECT department_id, COUNT(*) FROM Employee GROUP BY department_id;
22) Show departments where average salary is greater than 60,000
SELECT department_id FROM Employee GROUP BY department_id HAVING AVG(salary) > 60000;
23) Find department-wise maximum salary
SELECT department_id, MAX(salary) FROM Employee GROUP BY department_id;
24) Find department-wise minimum age
SELECT department_id, MIN(age) FROM Employee GROUP BY department_id;
25) Show departments having total salary greater than 5,00,000
SELECT department_id, SUM(salary) FROM Employee GROUP BY department_id HAVING SUM(salary) > 500000;
